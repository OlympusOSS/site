name: CD — Hera

on:
  push:
    tags: ['v*']
  workflow_dispatch:

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ghcr.io/olympusoss/hera

jobs:
  validate:
    runs-on: ubuntu-latest
    permissions:
      contents: read
    outputs:
      tag_version: ${{ steps.extract.outputs.tag_version }}
    steps:
      - name: Extract tag version
        id: extract
        run: |
          TAG_VERSION="${GITHUB_REF_NAME#v}"
          echo "tag_version=${TAG_VERSION}" >> "$GITHUB_OUTPUT"
          echo "Tag version: ${TAG_VERSION}"

      - name: Validate tag-commit ancestry (athena#121 AC1)
        # Uses GitHub API instead of git merge-base to avoid shallow clone issues.
        # Status must be "behind" (tag is ancestor of main) or "identical" (tag IS main HEAD).
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          STATUS=$(gh api repos/${{ github.repository }}/compare/main...${{ github.ref_name }} \
            --jq '.status')
          echo "Ancestry status: ${STATUS}"
          if [[ "${STATUS}" != "identical" && "${STATUS}" != "behind" ]]; then
            echo "::error::Tag ${{ github.ref_name }} is not reachable from main (status: ${STATUS}). Aborting deploy."
            exit 1
          fi

      - uses: actions/checkout@v6
        with:
          ref: ${{ github.ref_name }}

      - name: Validate tag-version consistency (athena#121 AC2)
        # Prevents orphan tags where package.json version doesn't match the tag.
        # This is the critical gate — catches tags pushed without version bumps.
        run: |
          PKG_VERSION=$(node -p "require('./package.json').version")
          TAG_VERSION="${GITHUB_REF_NAME#v}"
          echo "package.json version: ${PKG_VERSION}"
          echo "Tag version: ${TAG_VERSION}"
          if [[ "${PKG_VERSION}" != "${TAG_VERSION}" ]]; then
            echo "::error::Version mismatch: package.json=${PKG_VERSION}, tag=${TAG_VERSION}. Aborting deploy."
            exit 1
          fi

  version:
