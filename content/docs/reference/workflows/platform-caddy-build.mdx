name: Build and push custom Caddy image

# Builds the custom Caddy image (with mholt/caddy-ratelimit module) and
# publishes it to ghcr.io/olympusoss/caddy:latest and a sha-<short> tag.
#
# Required: SR-CADDY-1 (platform#22) â€” standard caddy:2-alpine does not include
# the caddy-ratelimit module needed for per-IP rate limiting on CIAM login routes.
#
# The sha-<short> tag provides a stable rollback reference for every build
# (SR-CADDY-4 / platform#49). The 'latest' tag is the production reference.
#
# After push, a verify job confirms the built image contains the
# http.handlers.rate_limit module (SR-CADDY-3 / platform#48). If the module
# is absent, the workflow fails and the broken image is flagged before deployment.
#
# Triggers:
#   - Push to main that modifies caddy/Containerfile (rebuilds when the image changes)
#   - Manual workflow_dispatch (force rebuild / rotate)

on:
  push:
    branches:
      - main
    paths:
      - 'caddy/Containerfile'
  workflow_dispatch:

concurrency:
  group: caddy-build
  cancel-in-progress: false

jobs:
  build-and-push:
    runs-on: ubuntu-latest

    permissions:
      contents: read
      packages: write

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Log in to GHCR
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      # Required for cross-arch builds: Buildx uses QEMU emulation to compile
      # the arm64 image on this amd64 runner. Without this, building for
      # linux/arm64 fails with "exec format error" when xcaddy runs 'go build'.
      - name: Set up QEMU
        uses: docker/setup-qemu-action@v3
        with:
          platforms: linux/amd64,linux/arm64

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build and push
        uses: docker/build-push-action@v6
        with:
          context: ./caddy
          file: .
