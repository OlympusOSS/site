name: Release ŌĆö Hera

# Changesets-powered release for Hera. Runs ONLY after CI completes
# successfully on main. On each run:
#   1. If there are pending changesets, consume them locally (version bump +
#      CHANGELOG update), commit directly to main, push the resulting
#      'v<version>' git tag.
#   2. If no pending changesets, this workflow is a no-op.
#
# The existing CD ŌĆö Hera workflow (.github/workflows/cd.yml) listens for
# 'v*' tags and handles the GHCR container build + deploy from there. This
# workflow does NOT publish to npm, Hera is '"private": true', so
# 'changeset publish' just creates the git tag locally and exits without
# hitting any registry.
#
# Avoids the PR-creation path entirely so it does not need the
# "Allow GitHub Actions to create and approve pull requests" repo setting.

on:
  workflow_run:
    workflows: ["CI ŌĆö Hera"]
    types: [completed]
    branches: [main]
  workflow_dispatch:

permissions:
  contents: write
  # 'actions: write' lets the workflow's GITHUB_TOKEN call the
  # workflow_dispatch API to fire 'CD ŌĆö Hera' on the new tag. Without
  # it the dispatch returns "HTTP 403: Resource not accessible by
  # integration" (default GITHUB_TOKEN permissions don't include
  # cross-workflow dispatch).
  actions: write

concurrency:
  group: release
  cancel-in-progress: false

jobs:
  release:
    # workflow_run fires on every completion (success/failure/cancelled).
    # Gate to success-only so a red CI doesn't bump the version.
    if: ${{ github.event_name == 'workflow_dispatch' || github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6
        with:
          fetch-depth: 0
          token: ${{ secrets.GITHUB_TOKEN }}
          # workflow_run runs against the default branch by default; pin to the
          # SHA that triggered CI so we tag the exact code that was tested.
          ref: ${{ github.event.workflow_run.head_sha || github.ref }}

      - uses: oven
