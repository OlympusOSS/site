name: CI — Hera

on:
  push:
    branches: [main]
    paths:
      - 'src/**'
      - 'test/**'
      - 'tests/**'
      - 'scripts/**'
      - 'package.json'
      - 'bun.lock'
      - 'Containerfile'
      - 'next.config.*'
      - 'playwright.config.*'
      - 'vitest.config.*'
      - '.github/workflows/**'
  pull_request:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read
  packages: read

env:
  # Deterministic fixtures for tests that read env. NOT used at runtime.
  ENCRYPTION_KEY: "AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA="
  SESSION_SIGNING_KEY: "AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA="
  HYDRA_ADMIN_URL: "http://localhost:4445"
  KRATOS_PUBLIC_URL: "http://localhost:4433"

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest
      - name: Configure npm for GitHub Packages
        run: |
          echo "@olympusoss:registry=https://npm.pkg.github.com" >> .npmrc
          echo "//npm.pkg.github.com/:_authToken=$\{\{ secrets.GITHUB_TOKEN \}\}" >> .npmrc
      - run: bun install
      - run: bun run typecheck
      - name: Run unit tests
        run: bun run test
      - name: Snapshot coverage gate (100%)
        run: bun run snapshot-coverage
      - name: Run unit tests with coverage (100/100/100/90 gate)
        run: bun run test:coverage
      - run: bun run build
      - name: Install Playwright browsers
        run: bunx playwright install --with-deps chromium
      - name: Run E2E tests
        run: bun run test:e2e
        env:
          CI: true
