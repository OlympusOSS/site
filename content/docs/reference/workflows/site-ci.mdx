## Source

```yaml
name: CI — Site

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read
  packages: read

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest
      - name: Configure npm for the @olympusoss scope
        # @olympusoss/canvas is published to npmjs.org by canvas's
        # release workflow; auth with NPM_TOKEN (org-scoped read-only).
        run: |
          echo "@olympusoss:registry=https://registry.npmjs.org" >> .npmrc
          echo "//registry.npmjs.org/:_authToken=${{ secrets.NPM_TOKEN }}" >> .npmrc
      - run: bun install --frozen-lockfile
      - run: bun run lint
      # 'next build' generates the fumadocs '.source/' runtime AND runs the
      # full TypeScript check against the produced source — same engine the
      # standalone 'bun run typecheck' script would invoke. Skipping the
      # separate typecheck step keeps '.source/' generation in the right
      # order: a bare 'tsc --noEmit' runs before MDX scaffolding exists and
      # fails to resolve './.source/server'.
      - run: bun run build
      - name: Install Playwright browsers
        run: bunx playwright install --with-deps chromium
      - name: Run E2E tests
        run: bun run test:e2e
        env:
          CI: true

```
