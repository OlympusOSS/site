name: Release — octl

# Changesets-powered release. Runs ONLY after the CI workflow completes
# successfully on main — avoids racing CI in parallel and never publishes a
# version that hasn't passed checks. On a successful CI run:
#   1. If there are pending changesets, consume them locally (version bump +
#      CHANGELOG update), commit directly to main, publish to npmjs.org.
#   2. If no pending changesets, this workflow is a no-op.
#
# Avoids the PR-creation path entirely so it does not need the
# "Allow GitHub Actions to create and approve pull requests" repo setting.

on:
  workflow_run:
    workflows: ["CI — octl"]
    types: [completed]
    branches: [main]
  workflow_dispatch:

permissions:
  contents: write
  # id-token: write enables npm provenance attestations on publish.
  id-token: write

concurrency:
  group: release
  cancel-in-progress: false

jobs:
  release:
    # workflow_run fires on every completion (success/failure/cancelled). Gate
    # to success-only here so a red CI doesn't trigger a publish attempt.
    if: ${{ github.event_name == 'workflow_dispatch' || github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6
        with:
          fetch-depth: 0
          token: ${{ secrets.GITHUB_TOKEN }}

          # SHA that triggered CI so we publish the exact code that was tested.
          ref: ${{ github.event.workflow_run.head_sha || github.ref }}
