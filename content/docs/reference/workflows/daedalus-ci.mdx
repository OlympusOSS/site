## Source

```yaml
name: CI — daedalus

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

permissions:
  contents: read
  packages: read

jobs:
  frontend:
    name: Frontend — build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6

      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest

      - name: Configure npm for GitHub Packages
        run: |
          echo "//npm.pkg.github.com/:_authToken=${{ secrets.GITHUB_TOKEN }}" >> .npmrc
          echo "@olympusoss:registry=https://npm.pkg.github.com" >> .npmrc

      - run: bun install --frozen-lockfile

      - name: Typecheck
        run: bunx tsc --noEmit

      - name: Build web bundle
        run: bun run build

      - name: Tests + coverage (100/100/100/90 gate)
        run: bun run test:coverage

      - name: Snapshot coverage gate (100%)
        run: bun run snapshot-coverage

  rust:
    name: Rust — check + clippy + fmt
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6

      - name: Install Rust toolchain
        run: |
          rustup toolchain install stable
          rustup component add clippy rustfmt

      - name: Install Tauri system deps
        run: |
          sudo apt-get update
          sudo apt-get install -y libwebkit2gtk-4.1-dev libappindicator3-dev librsvg2-dev patchelf

      - name: cargo fmt --check
        working-directory: src-tauri
        run: cargo fmt --all -- --check

      - name: cargo check
        working-directory: src-tauri
        run: cargo check --all-targets

      # Clippy runs without '-D warnings' for now — a handful of dead-code
      # warnings exist in screenshot.rs and mcp/tools.rs for future features.
      # Tighten to '-D warnings' once those code paths are wired up or removed.
      - name: cargo clippy
        working-directory: src-tauri
        run: cargo clippy --all-targets

```
