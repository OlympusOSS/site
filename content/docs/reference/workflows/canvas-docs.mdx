```yaml
name: Deploy docs to GitHub Pages

# Production docs deploys are driven by the Release workflow so the version
# baked into the bundle always matches the version that was just published.
# This workflow is kept for manual rebuilds via workflow_dispatch only.
on:
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: pages
  cancel-in-progress: false

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v6

      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: latest

      - name: Install Canvas deps (for type resolution)
        run: bun install --frozen-lockfile

      - name: Install docs deps
        working-directory: docs
        run: bun install --frozen-lockfile

      - name: Typecheck docs
        working-directory: docs
        run: bunx tsc --noEmit

      - name: Build docs
        working-directory: docs
        run: bun run build
        env:
          VITE_BASE_PATH: /canvas/

      - uses: actions/configure-pages@v5

      - uses: actions/upload-pages-artifact@v3
        with:
          path: docs/dist

  deploy:
    needs: build
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - uses: actions/deploy-pages@v4
        id: deployment

```
