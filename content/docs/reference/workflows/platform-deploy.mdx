name: Deploy

on:
  workflow_dispatch:
    inputs:
      environment:
        description: 'Target environment'
        required: true
        type: choice
        default: 'production'
        options:
          - production

concurrency:
  group: deploy-production
  cancel-in-progress: false

jobs:
  deploy:
    runs-on: ubuntu-latest

    env:
      SSH_OPTS: -i ~/.ssh/deploy_key -p ${{ vars.DEPLOY_SSH_PORT || 22 }} -o StrictHostKeyChecking=accept-new -o ConnectTimeout=30 -o ServerAliveInterval=30
      SSH_TARGET: ${{ vars.DEPLOY_USER || 'root' }}@${{ vars.DEPLOY_SERVER_IP }}
      DEPLOY_PATH: ${{ vars.DEPLOY_PATH || '/opt/olympusoss/prod' }}
      COMPOSE: podman compose -f compose.prod.yml

    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Configure SSH
        run: |
          mkdir -p ~/.ssh
          echo "${{ secrets.DEPLOY_SSH_KEY }}" > ~/.ssh/deploy_key
          chmod 600 ~/.ssh/deploy_key
          ssh-keyscan -T 10 -p ${{ vars.DEPLOY_SSH_PORT || 22 }} ${{ vars.DEPLOY_SERVER_IP }} >> ~/.ssh/known_hosts 2>/dev/null || true

      - name: Verify SSH connectivity (retry up to 5 minutes)
        run: |
          MAX_ATTEMPTS=10
          WAIT=30
          for i in $(seq 1 $MAX_ATTEMPTS); do
            echo "[Attempt $i/$MAX_ATTEMPTS] Connecting to ${{ env.SSH_TARGET }}..."
            if ssh ${{ env.SSH_OPTS }} ${{ env.SSH_TARGET }} "mkdir -p ${{ env.DEPLOY_PATH }}" 2>/dev/null; then
              echo "SSH connection verified."
              exit 0
            fi
            if [ "$i" -lt "$MAX_ATTEMPTS" ]; then
              echo "Connection failed. Waiting ${WAIT}s before retry..."
              sleep $WAIT
            fi
          done
          echo "::error::SSH connection failed after $MAX_ATTEMPTS attempts ($(($MAX_ATTEMPTS * $WAIT))s). Server may be unreachable."
          exit 1
